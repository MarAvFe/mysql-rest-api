var express = require("express");


/**
 * @function usuario
 *@param {int(11)} ID - Key: PRI
 *@param {varchar(50)} Usuario
 *@param {varchar(100)} NombreCompleto
 *@param {varchar(50)} Clave
 *@param {varchar(50)} EmailContacto
 */


exports.router = function (connection) {
	var router = express.Router();

	var create = express.Router();
	create.post("/add", function (req, res) {
		connection.query("INSERT INTO usuario (ID, Usuario, NombreCompleto, Clave, EmailContacto) VALUES( :V_ID, :V_Usuario, :V_NombreCompleto, :V_Clave, :V_EmailContacto )", req.body, res);
	});
 	router.use("/usuario", create);

	var read = express.Router();
	read.post("/get", function (req, res) {
		connection.query("SELECT :C FROM usuario WHERE :OU", req.body, res);
	});
 	router.use("/usuario", read);

	var update = express.Router();
	update.post("/update", function (req, res) {
		connection.query("UPDATE usuario SET :OC WHERE :OF", req.body, res);
	});
 	router.use("/usuario", update);

	var deletePath = express.Router();
	deletePath.post("/delete", function (req, res) {
		connection.query("DELETE FROM usuario WHERE :OR", req.body, res);
	});
 	router.use("/usuario", deletePath);

	return router;
}
