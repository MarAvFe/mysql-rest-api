var express = require("express");


/**
 * @function pais
 *@param {int(11)} Id - Key: PRI
 *@param {varchar(50)} Nombre
 *@param {varchar(2)} Iso
 *@param {varchar(3)} Iso3
 *@param {int(11)} CodigoNumerico
 *@param {int(11)} CodigoTelefono
 */


exports.router = function (connection) {
	var router = express.Router();

	var create = express.Router();
	create.post("/add", function (req, res) {
		connection.query("INSERT INTO pais (Id, Nombre, Iso, Iso3, CodigoNumerico, CodigoTelefono) VALUES( :V_Id, :V_Nombre, :V_Iso, :V_Iso3, :V_CodigoNumerico, :V_CodigoTelefono )", req.body, res);
	});
 	router.use("/pais", create);

	var read = express.Router();
	read.post("/get", function (req, res) {
		connection.query("SELECT :C FROM pais WHERE :OU", req.body, res);
	});
 	router.use("/pais", read);

	var update = express.Router();
	update.post("/update", function (req, res) {
		connection.query("UPDATE pais SET :OC WHERE :OF", req.body, res);
	});
 	router.use("/pais", update);

	var deletePath = express.Router();
	deletePath.post("/delete", function (req, res) {
		connection.query("DELETE FROM pais WHERE :OR", req.body, res);
	});
 	router.use("/pais", deletePath);

	return router;
}
