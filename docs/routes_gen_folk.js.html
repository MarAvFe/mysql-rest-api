var express = require("express");


/**
 * @function folk
 *@param {int(11)} rate
 *@param {decimal(25,0)} spec
 *@param {tinyint(1)} peace
 */


exports.router = function (connection) {
	var router = express.Router();

	var create = express.Router();
	create.post("/add", function (req, res) {
		connection.query("INSERT INTO folk (rate, spec, peace) VALUES( :V_rate, :V_spec, :V_peace )", req.body, res);
	});
 	router.use("/folk", create);

	var read = express.Router();
	read.post("/get", function (req, res) {
		connection.query("SELECT :C FROM folk WHERE :OU", req.body, res);
	});
 	router.use("/folk", read);

	var update = express.Router();
	update.post("/update", function (req, res) {
		connection.query("UPDATE folk SET :OC WHERE :OF", req.body, res);
	});
 	router.use("/folk", update);

	var deletePath = express.Router();
	deletePath.post("/delete", function (req, res) {
		connection.query("DELETE FROM folk WHERE :OR", req.body, res);
	});
 	router.use("/folk", deletePath);

	return router;
}
