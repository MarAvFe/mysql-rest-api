var express = require("express");


/**
 * @function respuesta
 *@param {int(11)} ID - Key: PRI
 *@param {int(11)} Respuesta01
 *@param {int(11)} Respuesta02
 *@param {int(11)} Respuesta03
 *@param {int(11)} Respuesta04
 *@param {int(11)} Respuesta05
 *@param {datetime} FechaRespuesta
 *@param {varchar(1)} GeneroID - Key: MUL
 *@param {varchar(100)} Campo01
 *@param {varchar(100)} Campo02
 *@param {tinyint(1)} RangoEdad
 *@param {int(11)} PymeID - Key: MUL
 */


exports.router = function (connection) {
	var router = express.Router();

	var create = express.Router();
	create.post("/add", function (req, res) {
		connection.query("INSERT INTO respuesta (ID, Respuesta01, Respuesta02, Respuesta03, Respuesta04, Respuesta05, FechaRespuesta, GeneroID, Campo01, Campo02, RangoEdad, PymeID) VALUES( :V_ID, :V_Respuesta01, :V_Respuesta02, :V_Respuesta03, :V_Respuesta04, :V_Respuesta05, :V_FechaRespuesta, :V_GeneroID, :V_Campo01, :V_Campo02, :V_RangoEdad, :V_PymeID )", req.body, res);
	});
 	router.use("/respuesta", create);

	var read = express.Router();
	read.post("/get", function (req, res) {
		connection.query("SELECT :C FROM respuesta WHERE :OU", req.body, res);
	});
 	router.use("/respuesta", read);

	var update = express.Router();
	update.post("/update", function (req, res) {
		connection.query("UPDATE respuesta SET :OC WHERE :OF", req.body, res);
	});
 	router.use("/respuesta", update);

	var deletePath = express.Router();
	deletePath.post("/delete", function (req, res) {
		connection.query("DELETE FROM respuesta WHERE :OR", req.body, res);
	});
 	router.use("/respuesta", deletePath);

	return router;
}
