var express = require("express");


/**
 * @function pyme
 *@param {int(11)} Id - Key: PRI
 *@param {char(100)} NombreComercio
 *@param {int(11)} EstadoID - Key: MUL
 *@param {int(11)} SectorID - Key: MUL
 *@param {tinyint(4)} AnnoInicioOperaciones
 *@param {char(50)} NumeroTelefono
 *@param {char(200)} Direccion
 *@param {bit(1)} EsActiva
 *@param {bit(1)} EsNegocioFamiliar
 *@param {blob} Logo
 *@param {char(3)} ExtensionLogo
 *@param {datetime} FechaCreacion
 *@param {datetime} FechaUltimaActualizacion
 *@param {bit(1)} EsFacebookAppInstalado
 *@param {int(11)} UsuarioID - Key: MUL
 *@param {varchar(1)} GeneroPropietarioID - Key: MUL
 *@param {varchar(50)} CedJuridica
 */


exports.router = function (connection) {
	var router = express.Router();

	var create = express.Router();
	create.post("/add", function (req, res) {
		connection.query("INSERT INTO pyme (Id, NombreComercio, EstadoID, SectorID, AnnoInicioOperaciones, NumeroTelefono, Direccion, EsActiva, EsNegocioFamiliar, Logo, ExtensionLogo, FechaCreacion, FechaUltimaActualizacion, EsFacebookAppInstalado, UsuarioID, GeneroPropietarioID, CedJuridica) VALUES( :V_Id, :V_NombreComercio, :V_EstadoID, :V_SectorID, :V_AnnoInicioOperaciones, :V_NumeroTelefono, :V_Direccion, :V_EsActiva, :V_EsNegocioFamiliar, :V_Logo, :V_ExtensionLogo, :V_FechaCreacion, :V_FechaUltimaActualizacion, :V_EsFacebookAppInstalado, :V_UsuarioID, :V_GeneroPropietarioID, :V_CedJuridica )", req.body, res);
	});
 	router.use("/pyme", create);

	var read = express.Router();
	read.post("/get", function (req, res) {
		connection.query("SELECT :C FROM pyme WHERE :OU", req.body, res);
	});
 	router.use("/pyme", read);

	var update = express.Router();
	update.post("/update", function (req, res) {
		connection.query("UPDATE pyme SET :OC WHERE :OF", req.body, res);
	});
 	router.use("/pyme", update);

	var deletePath = express.Router();
	deletePath.post("/delete", function (req, res) {
		connection.query("DELETE FROM pyme WHERE :OR", req.body, res);
	});
 	router.use("/pyme", deletePath);

	return router;
}
