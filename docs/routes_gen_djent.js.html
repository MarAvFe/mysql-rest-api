var express = require("express");


/**
 * @function djent
 *@param {text} doesit
 *@param {text} ofcourseitdoes
 *@param {int(10)} id
 */


exports.router = function (connection) {
	var router = express.Router();

	var create = express.Router();
	create.post("/add", function (req, res) {
		connection.query("INSERT INTO djent (doesit, ofcourseitdoes, id) VALUES( :V_doesit, :V_ofcourseitdoes, :V_id )", req.body, res);
	});
 	router.use("/djent", create);

	var read = express.Router();
	read.post("/get", function (req, res) {
		connection.query("SELECT :C FROM djent WHERE :OU", req.body, res);
	});
 	router.use("/djent", read);

	var update = express.Router();
	update.post("/update", function (req, res) {
		connection.query("UPDATE djent SET :OC WHERE :OF", req.body, res);
	});
 	router.use("/djent", update);

	var deletePath = express.Router();
	deletePath.post("/delete", function (req, res) {
		connection.query("DELETE FROM djent WHERE :OR", req.body, res);
	});
 	router.use("/djent", deletePath);

	return router;
}
