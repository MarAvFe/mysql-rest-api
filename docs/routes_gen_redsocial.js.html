var express = require("express");


/**
 * @function redsocial
 *@param {int(11)} TipoRedSocialID - Key: PRI
 *@param {varchar(300)} Comentario
 *@param {varchar(300)} InformacionContacto
 *@param {int(11)} PymeID - Key: PRI
 *@param {varchar(300)} Link
 */


exports.router = function (connection) {
	var router = express.Router();

	var create = express.Router();
	create.post("/add", function (req, res) {
		connection.query("INSERT INTO redsocial (TipoRedSocialID, Comentario, InformacionContacto, PymeID, Link) VALUES( :V_TipoRedSocialID, :V_Comentario, :V_InformacionContacto, :V_PymeID, :V_Link )", req.body, res);
	});
 	router.use("/redsocial", create);

	var read = express.Router();
	read.post("/get", function (req, res) {
		connection.query("SELECT :C FROM redsocial WHERE :OU", req.body, res);
	});
 	router.use("/redsocial", read);

	var update = express.Router();
	update.post("/update", function (req, res) {
		connection.query("UPDATE redsocial SET :OC WHERE :OF", req.body, res);
	});
 	router.use("/redsocial", update);

	var deletePath = express.Router();
	deletePath.post("/delete", function (req, res) {
		connection.query("DELETE FROM redsocial WHERE :OR", req.body, res);
	});
 	router.use("/redsocial", deletePath);

	return router;
}
